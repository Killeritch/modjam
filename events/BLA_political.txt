add_namespace = BLA_political

#Albania wants more influence
country_event = {
	id = BLA_political.1
	title = BLA_political.1.t
	desc = BLA_political.1.d
	picture = GFX_report_event_generic_usa_treaty
	
	fire_only_once = yes
	
	trigger = {
		TAG = BLA
		date > 1936.3.1
	}
	
	option = {
		name = BLA_political.1.a
		ai_chance = { factor = 60 }
		add_timed_idea = { idea = BLA_AlbInf days = 180 }
	}
	
	option = {
		name = BLA_political.1.b
		ai_chance = { factor = 40 }
		add_national_unity = -0.05
	}
}

#Intervening in the Greek Civil War
country_event = {
	id = BLA_political.2
	title = BLA_political.2.t
	desc = BLA_political.2.d
	picture = GFX_report_vienna_award_hungary
	
	fire_only_once = yes
	
	trigger = {
		TAG = BLA
		DSE = { has_war_with = GTU }
		NOT = { BLA = { has_government = totalist } }
	}
	
	option = {
		name = BLA_political.2.a
		add_political_power = 25
		ai_chance = { factor = 50 }
	}
	
	option = {
		name = BLA_political.2.b
		add_political_power = -25
		GTU = { country_event = { days = 1 id = BLA_political.3 } }
		ai_chance = { factor = 35 }
		add_manpower = -500
	}
	
	option = {
		name = BLA_political.2.c
		add_political_power = -75
		GTU = { country_event = { days = 1 id = BLA_political.4 } }
		ai_chance = { factor = 15 }
		add_manpower = -3000
	}
}
